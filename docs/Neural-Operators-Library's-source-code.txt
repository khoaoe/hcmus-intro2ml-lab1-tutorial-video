 3 | import torch
 4 | import torch.nn as nn
 5 | 
 6 | 
 7 | from .base_model import BaseModel
 8 | 
 9 | 
10 | class UQNO(BaseModel, name="UQNO"):
11 |     """Uncertainty Quantification Neural Operator
12 |     
13 |     General N-dim (alpha, delta) Risk-Controlling Neural Operator, as described in [1]_.
14 | 
15 |     The UQNO is trained to map input functions to a residual function
16 |     E(a, x) that describes the predicted error between the ground truth
17 |     and the outputs of a trained model. E(a, x) is then used in combination
18 |     with a calibrated scaling factor to predict calibrated uncertainty bands
19 |     around the predictions of the trained model.
20 | 
21 |     Parameters
22 |     ----------
23 |     base_model : nn.Module
24 |         Pre-trained solution operator. Determined by the problem.
25 |     residual_model : nn.Module, optional
26 |         Architecture to train as the UQNO's quantile model.
27 |         If None, a deep copy of base_model is used. Default: None
28 | 
29 |     References
30 |     ----------
31 |     .. [1] :
32 | 
33 |     Ma, Z., Pitt, D., Azizzadenesheli, K., and Anandkumar, A. (2024).
34 |         "Calibrated Uncertainty Quantification for Operator Learning
35 |         via Conformal Prediction". TMLR, https://openreview.net/pdf?id=cGpegxy12T.
36 |     """
37 | 
38 |     def __init__(
39 |         self, base_model: nn.Module, residual_model: nn.Module = None, **kwargs
40 |     ):
41 |         super().__init__()
42 | 
43 |         self.base_model = base_model
44 |         if residual_model is None:
45 |             residual_model = deepcopy(base_model)
46 |         self.residual_model = residual_model
47 | 
48 |     def forward(self, *args, **kwargs):
49 |         """
50 |         Forward pass returns the solution u(a,x)
51 |         and the uncertainty ball E(a,x) as a pair
52 |         for pointwise quantile loss
53 |         """
54 |         self.base_model.eval()  # base-model weights are frozen
55 |         # another way to handle this would be to use LoRA, or similar
56 |         # ie freeze the  weights, and train a low-rank matrix of weight perturbations
57 |         with torch.no_grad():
58 |             solution = self.base_model(*args, **kwargs)
59 |         quantile = self.residual_model(*args, **kwargs)
60 |         return (solution, quantile)
61 |

12 | def CGELU(x: torch.Tensor):
13 |     """Complex GELU activation function
14 |     Follows the formulation of CReLU from [1]_.
15 |     Applies GELU to real and imaginary parts of the input
16 |     separately, then combine as complex number
17 | 
18 | 
19 |     Parameters
20 |     -----------
21 |     x : torch.tensor (dtype=complex)
22 |         pre-activation inputs
23 | 
24 |     References
25 |     ----------
26 |     .. [1] :
27 | 
28 |     Trabelsi, C., et al. (2018). "Deep Complex Networks".
29 |         ICLR 2018, https://openreview.net/pdf?id=H1T2hmZAb.
30 |     """

38 |     Args:
39 |         x: complex tensor
40 |     """

45 |     """Complex-valued tanh stabilizer
46 |     Apply ctanh is real and imag part of the input separately, then combine as complex number
47 |     Args:
48 |         x: complex tensor
49 |     """
50 |     return torch.tanh(x.real).type(torch.cfloat) + 1j * torch.tanh(x.imag).type(

10 | def einsum_complexhalf_two_input(eq, a, b):
11 |     """
12 |     Compute (two-input) einsum for complexhalf tensors.
13 |     Because torch.einsum currently does not support complex32 (complexhalf) types.
14 |     The inputs and outputs are the same as in torch.einsum
15 |     """
16 |     assert len(eq.split(",")) == 2, "Equation must have two inputs."
17 | 
18 |     # cast both tensors to "view as real" form, and half precision
19 |     a = torch.view_as_real(a)
20 |     b = torch.view_as_real(b)
21 |     a = a.half()
22 |     b = b.half()
23 | 
24 |     # create a new einsum equation that takes into account "view as real" form
25 |     input_output = eq.split("->")
26 |     new_output = "xy" + input_output[1]
27 |     input_terms = input_output[0].split(",")
28 |     new_inputs = [input_terms[0] + "x", input_terms[1] + "y"]
29 |     new_eqn = new_inputs[0] + "," + new_inputs[1] + "->" + new_output
30 | 
31 |     # convert back to complex form
32 |     tmp = tl.einsum(new_eqn, a, b)
33 |     res = torch.stack(
34 |         [tmp[0, 0, ...] - tmp[1, 1, ...], tmp[1, 0, ...] + tmp[0, 1, ...]], dim=-1
35 |     )
36 |     return torch.view_as_complex(res)

40 |     """
41 |     Compute einsum for complexhalf tensors.
42 |     Because torch.einsum currently does not support complex32 (complexhalf) types.
43 |     The inputs and outputs are the same as in torch.einsum
44 |     """
45 |     if len(args) == 2:
46 |         # if there are two inputs, it is faster to call this method
47 |         return einsum_complexhalf_two_input(eq, *args)

51 |     partial_eqns = [contraction_info[2] for contraction_info in path_info.contraction_list]
52 | 
53 |     # create a dict of the input tensors by their label in the einsum equation
54 |     tensors = {}

56 |     output_label = eq.split("->")[1]
57 |     tensors = dict(zip(input_labels, args))

  8 |     
  9 |     This is the finite difference convolution layer introduced in [1]_.
 10 |     
 11 |     It computes a finite difference convolution on a regular grid,
 12 |     which converges to a directional derivative as the grid is refined.
 13 | 
 14 |     Parameters
 15 |     ----------
 16 |     in_channels : int
 17 |         Number of in_channels
 18 |     out_channels : int

 24 |     groups : int, optional
 25 |         Splitting number of channels, by default 1
 26 |     padding : literal {'periodic', 'replicate', 'reflect', 'zeros'}, optional
 27 |         Mode of padding to use on input. Options: 'periodic', 'replicate', 'reflect', 'zeros', by default 'periodic'
 28 |         See `torch.nn.functional.padding`.
 29 | 
 30 |     References
 31 |     ----------
 32 |     .. [1] : Liu-Schiaffini, M., et al. (2024). "Neural Operators with
 33 |         Localized Integral and Differential Kernels".
 34 |         ICML 2024, https://arxiv.org/abs/2402.16845.
 35 | 
 36 |     """
 37 | 
 38 |     def __init__(
 39 |         self,
 40 |         in_channels,
 41 |         out_channels,
 42 |         n_dim,
 43 |         kernel_size=3,
 44 |         groups=1,
 45 |         padding="periodic",
 46 |     ):
 47 |         super().__init__()
 48 |         conv_module = getattr(nn, f"Conv{n_dim}d")
 49 |         self.F_conv_module = getattr(F, f"conv{n_dim}d")
 50 |         self.conv_function = getattr(F, f"conv{n_dim}d")
 51 | 
 52 |         assert kernel_size % 2 == 1, "Kernel size should be odd"
 53 |         self.kernel_size = kernel_size
 54 | 
 55 |         self.in_channels = in_channels
 56 |         self.groups = groups
 57 |         self.n_dim = n_dim
 58 | 
 59 |         if padding == "periodic":
 60 |             self.padding_mode = "circular"
 61 |         elif padding == "replicate":
 62 |             self.padding_mode = "replicate"
 63 |         elif padding == "reflect":
 64 |             self.padding_mode = "reflect"
 65 |         elif padding == "zeros":
 66 |             self.padding_mode = "zeros"
 67 |         else:
 68 |             raise NotImplementedError("Desired padding mode is not currently supported")
 69 |         self.pad_size = kernel_size // 2
 70 | 
 71 |         self.conv = conv_module(
 72 |             in_channels,
 73 |             out_channels,
 74 |             kernel_size=kernel_size,
 75 |             padding="same",
 76 |             padding_mode=self.padding_mode,
 77 |             bias=False,
 78 |             groups=groups,
 79 |         )
 80 |         self.weight = self.conv.weight

 82 |     def forward(self, x, grid_width):
 83 |         """FiniteDifferenceConvolution's forward pass. Alternatively,
 84 |         one could center the conv kernel by subtracting the mean pointwise
 85 |         in the kernel: ``conv(x, kernel - mean(kernel)) / grid_width``
 86 | 
 87 |         Parameters
 88 |         ----------
 89 |         x : torch.tensor
 90 |             input tensor, shape (batch, in_channels, d_1, d_2, ...d_n)
 91 |         grid_width : float
 92 |             discretization size of input grid
 93 |         """
 94 |         conv = self.conv(x)
 95 |         conv_sum = torch.sum(
 96 |             self.conv.weight,
 97 |             dim=tuple([i for i in range(2, 2 + self.n_dim)]),
 98 |             keepdim=True,
 99 |         )
100 |         conv_sum = self.conv_function(x, conv_sum, groups=self.groups)
101 |         return (conv - conv_sum) / grid_width

 17 | import torch

 19 | import matplotlib.pyplot as plt
 20 | import matplotlib.animation as animation

 28 | # .. raw:: html
 29 | #
 30 | #    <div style="margin-top: 3em;"></div>
 31 | #
 32 | # Defining our problem
 33 | # --------------------
 34 | # We aim to solve the 2D diffusion advection equation:
 35 | #
 36 | # .. math::
 37 | #    \frac{\partial u}{\partial t} + c_x \frac{\partial u}{\partial x} + c_y \frac{\partial u}{\partial y} = \nu \left(\frac{\partial^2 u}{\partial x^2} + \frac{\partial^2 u}{\partial y^2}\right) + f(x,y,t)
 38 | #
 39 | # Where :math:`f(x,y,t)` is a source term and :math:`c_x` and :math:`c_y` are advection speeds in x and y.
 40 | # We set simulation parameters below:
 41 | 
 42 | ## Simulation parameters
 43 | Lx, Ly = 2.0, 2.0  # Domain lengths
 44 | nx, ny = 64, 64  # Grid resolution
 45 | T = 1.6  # Total simulation time
 46 | dt = 0.001  # Time step
 47 | nu = 0.02  # diffusion coefficient
 48 | cx, cy = 1.0, 0.6  # advection speeds
 49 | 
 50 | ## Create grid
 51 | X = torch.linspace(0, Lx, nx, device=device).repeat(ny, 1).T
 52 | Y = torch.linspace(0, Ly, ny, device=device).repeat(nx, 1)
 53 | dx = Lx / (nx - 1)
 54 | dy = Ly / (ny - 1)
 55 | nt = int(T / dt)
 56 | 
 57 | ## Initialize finite difference operator
 58 | fd = FiniteDiff(dim=2, h=(dx, dy))
 59 | 
 60 | 
 61 | ## Initial condition and source term
 62 | u = (
 63 |     -torch.sin(2 * np.pi * Y) * torch.cos(2 * np.pi * X)
 64 |     + 0.3 * torch.exp(-((X - 0.75) ** 2 + (Y - 0.5) ** 2) / 0.02)
 65 |     - 0.3 * torch.exp(-((X - 1.25) ** 2 + (Y - 1.5) ** 2) / 0.02)
 66 | ).to(device)

 70 |     return 0.2 * torch.sin(3 * np.pi * X) * torch.cos(3 * np.pi * Y) * torch.cos(4 * np.pi * t)

 74 | #
 75 | #    <div style="margin-top: 3em;"></div>
